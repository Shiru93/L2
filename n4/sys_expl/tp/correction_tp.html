Exercice — Clone de echo

mon_echo.c ✓ Correct

Reproduire la commande echo : afficher tous les arguments passés en ligne de commande séparés par des espaces, suivi d'un retour à la ligne.

Code commenté ligne par ligne
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[]) {

    /* argv[0] = nom du programme (ex: "./mon_echo")
     * argv[1], argv[2]... = les vrais arguments
     * On avance argv pour sauter argv[0] */
    argv++;  // argv pointe maintenant sur argv[1]

    while (*argv != NULL) {       // tant qu'il reste un argument

        while (**argv != '\0') {  // parcourir les caractères du mot
            putchar(**argv);    // afficher le caractère courant
            (*argv)++;           // avancer dans le mot (char suivant)
        }

        putchar(' ');   // espace entre les mots
        argv++;         // passer au mot suivant
    }

    putchar('\n');  // retour à la ligne final
    return 0;
}
ConceptExplication
argv++Avance le pointeur argv pour ignorer argv[0] (nom du programme). Maintenant *argv == argv[1], le premier vrai argument.
*argv != NULLLe tableau argv est terminé par un pointeur NULL. C'est la sentinelle de fin.
**argvDouble déréférencement : *argv = pointeur vers le mot (char*), **argv = caractère courant de ce mot.
(*argv)++Les parenthèses sont essentielles ! Sans elles, argv++ avancerait le tableau, pas le pointeur vers le char.

Exercice — Clone de ls

exo1.c · mon_ls.c ✓ Correct

exo1.c : version basique de ls. mon_ls.c : version avec options -a (tout afficher, y compris fichiers cachés) et -i (afficher l'inode).

Fonctions clés
/* opendir / readdir / closedir : API POSIX pour lister un répertoire */

void ls(const char *chemin, int options) {
    DIR *d = opendir(chemin);  // ouvre le répertoire → flux de lecture
    struct dirent *ent;

    if (!d) { perror("opendir"); return; }

    while ((ent = readdir(d))) {  // lit l'entrée suivante (NULL si fin)
        switch (options) {
        case 3:  // -a et -i : afficher tout + numéro d'inode
            printf("%lu : %s\n", (unsigned long)ent->d_ino, ent->d_name);
            break;
        case 2:  // -a seul : afficher tout (y compris .hidden)
            printf("%s\n", ent->d_name);
            break;
        case 1:  // -i seul : inode, mais cacher les . et ..
            if (ent->d_name[0] != '.')
                printf("%lu : %s\n", (unsigned long)ent->d_ino, ent->d_name);
            break;
        case 0:  // sans option : cacher les . et ..
            if (ent->d_name[0] != '.')
                printf("%s\n", ent->d_name);
            break;
        }
    }
    closedir(d);  // libère les ressources !
}
FonctionRôle
opendir(path)Ouvre un flux de lecture vers un répertoire. Retourne NULL en cas d'erreur (chemin inexistant, permission refusée...)
readdir(d)Retourne le prochain struct dirent (une entrée du répertoire). Retourne NULL quand il n'y en a plus.
ent->d_nameNom de l'entrée (fichier ou dossier) sous forme de chaîne.
ent->d_inoNuméro d'inode de l'entrée (identifiant unique sur le système de fichiers).
closedir(d)Ferme le flux. Toujours appeler pour éviter les fuites de descripteurs.

Exercice — Clone de rev (inverser les lignes)

rev.c ✓ Correct

Lire stdin caractère par caractère, accumuler les chars dans un buffer dynamique. Quand on rencontre \n, afficher le buffer à l'envers et recommencer. Le buffer grandit avec realloc si nécessaire.

Code annoté
int taille_max = 80;       // taille initiale du buffer
int index = 0;             // nombre de chars dans le buffer

char *buffer = malloc(sizeof(char) * taille_max);
if (!buffer) { perror("malloc"); return 1; }

while ((c = getchar()) != EOF) {

    if (c == '\n') {
        // Afficher le buffer à l'envers
        for (int i = index - 1; i >= 0; i--)
            putchar(buffer[i]);
        putchar('\n');
        index = 0;  // réinitialiser (on ne réalloue pas, on réutilise)

    } else {
        if (index >= taille_max) {  // buffer plein → agrandir
            taille_max *= 2;       // doubler la capacité (stratégie amortie O(1))
            char *nouveau = realloc(buffer, taille_max * sizeof(char));
            if (!nouveau) {
                perror("realloc");
                free(buffer);  // libérer l'ancien buffer !
                return 1;
            }
            buffer = nouveau;  // realloc peut déplacer la mémoire
        }
        buffer[index++] = c;  // stocker le char
    }
}
free(buffer);  // libérer à la fin
⚠️ Piège classique : Ne jamais faire buffer = realloc(buffer, ...) directement. Si realloc retourne NULL (échec), l'ancien buffer est perdu et non libéré → fuite mémoire. Toujours utiliser une variable temporaire (nouveau).

Exercice 2 — Mousse au chocolat parallèle (avec fork)

exo2.c ✓ Correct

Exploiter le parallélisme : certaines tâches sont indépendantes et peuvent se faire en même temps. L'arbre de fork crée des processus enfants pour les tâches parallélisables.

Architecture des processus : PÈRE (main) ├── fork() → FILS_1 │ ├── fork() ×6 → petits-fils (separer_blanc_jaune ×6 en parallèle) │ ├── wait() ×6 (attendre tous les petits-fils) │ └── battre_blancs_neige() ← après que tous les blancs soient séparés │ └── fork() → FILS_2 ├── casser_chocolat() └── faire_fondre_chocolat() ← en parallèle avec FILS_1 PÈRE : wait() × 2 (attendre FILS_1 et FILS_2) → melanger_jaunes_choco() → incorporer_blancs_neige() ← en dernier, dépend des deux
Code commenté
int main(void) {
    int i;

    switch(fork()) {        // 1er fork : créer FILS_1 (blancs)
    case -1:
        perror("fork"); return 1;

    case 0:                  // ← FILS_1 : s'occupe des blancs
        for (i = 0; i < 6; ++i) {
            switch(fork()) {  // fork 6 petits-fils pour séparer les blancs
            case -1: perror("fork"); return 1;
            case 0:            // ← petit-fils : sépare 1 blanc
                separer_blanc_jaune();
                return 0;     // IMPORTANT : le fils doit terminer !
            }
        }
        while(wait(NULL) != -1); // attendre tous les petits-fils
        battre_blancs_neige();
        return 0;             // FILS_1 termine

    default:                 // ← PÈRE : fork un 2ème fils pour le chocolat
        switch(fork()) {    // 2ème fork : FILS_2
        case -1: perror("fork"); return 1;
        case 0:              // ← FILS_2 : chocolat
            casser_chocolat();
            faire_fondre_chocolat();
            return 0;       // FILS_2 termine
        }
    }

    // PÈRE : attendre la fin des 2 fils
    while(wait(NULL) != -1);

    // Tout est prêt → finaliser
    melanger_jaunes_choco();
    incorporer_blancs_neige();
    printf("PID %d : la mousse est terminée !\n", getpid());
    return 0;
}
Appel systèmeRôle
fork()Crée une copie exacte du processus courant. Retourne 0 au fils, le PID du fils au père, -1 en cas d'erreur.
wait(NULL)Bloque jusqu'à la terminaison d'un fils quelconque. Retourne son PID, ou -1 s'il n'y a plus de fils.
getpid()Retourne le PID du processus courant.
return 0 (dans le fils)Indispensable ! Sans ça, le fils continue d'exécuter le code du père → comportement indéfini.

Exercice — mon_stat (afficher les métadonnées d'un fichier)

mon_stat.c ✗ À compléter (fichier vide)

Utiliser l'appel système stat() pour afficher les informations d'un fichier (taille, permissions, inode, dates...) comme la commande stat.

Correction complète
#include <stdint.h>
#include <stdio.h>
#include <stdlib.h>
#include <sys/stat.h>
#include <sys/sysmacros.h>  // major(), minor()
#include <time.h>

int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(stderr, "Usage: %s <fichier>\n", argv[0]);
        return 1;
    }

    struct stat s;
    if (stat(argv[1], &s) == -1) {
        perror("stat");
        return 2;
    }

    printf("Fichier    : %s\n", argv[1]);
    printf("Inode      : %lu\n", (unsigned long)s.st_ino);
    printf("Liens      : %lu\n", (unsigned long)s.st_nlink);
    printf("UID        : %u\n", s.st_uid);
    printf("GID        : %u\n", s.st_gid);
    printf("Taille     : %ld octets\n", (long)s.st_size);
    printf("Bloc size  : %ld\n", (long)s.st_blksize);
    printf("Blocs      : %lld\n", (long long)s.st_blocks);

    /* Permissions : st_mode contient type + droits */
    printf("Mode       : %04o\n", s.st_mode & 07777);

    /* Type de fichier */
    if      (S_ISREG(s.st_mode))  printf("Type       : fichier régulier\n");
    else if (S_ISDIR(s.st_mode))  printf("Type       : répertoire\n");
    else if (S_ISLNK(s.st_mode))  printf("Type       : lien symbolique\n");
    else if (S_ISCHR(s.st_mode))  printf("Type       : périphérique char\n");
    else if (S_ISBLK(s.st_mode))  printf("Type       : périphérique bloc\n");

    /* Dates (ctime convertit time_t en chaîne lisible) */
    printf("Modifié    : %s", ctime(&s.st_mtime));
    printf("Accédé     : %s", ctime(&s.st_atime));
    printf("Changé     : %s", ctime(&s.st_ctime));

    return 0;
}
ChampSignification
st_inoNuméro d'inode (identifiant unique du fichier sur le FS)
st_modeType du fichier + permissions (bits rwxrwxrwx) — utiliser S_ISREG(), S_ISDIR()... pour le type
st_nlinkNombre de liens physiques pointant vers cet inode
st_sizeTaille en octets (pour les fichiers réguliers)
st_mtimeDate de dernière modification du contenu
st_atimeDate de dernier accès (lecture)
st_ctimeDate de dernier changement des métadonnées (chmod, rename...)

Exercice — Calcul de π par la méthode des trapèzes

pi_trapèzes.c ⚠️ Version séquentielle — à paralléliser avec pthreads

Calcule π en intégrant f(x) = √(1−x²) sur [0,1] par la méthode des trapèzes. La valeur est 4 × ∫₀¹ √(1−x²) dx = π. Version parallèle avec N threads qui se partagent les tranches.

Version parallèle avec pthreads
#include <stdio.h>
#include <math.h>
#include <pthread.h>
#include <time.h>

#ifndef N_TRANCHES
#define N_TRANCHES (1 << 27)
#endif
#define PAS (1.0 / N_TRANCHES)
#ifndef NB_THREADS
#define NB_THREADS 4
#endif

struct tranche { long debut; long fin; double resultat; };

double f(double x) { return sqrt(1 - x * x); }

/* Chaque thread calcule la somme de trapèzes pour sa tranche */
void *calculer_tranche(void *arg) {
    struct tranche *t = arg;
    double res = 0.0;
    for (long i = t->debut; i < t->fin; i++) {
        double x = i * PAS;
        res += (f(x) + f(x + PAS)) / 2.0;
    }
    t->resultat = res;
    return NULL;
}

int main(void) {
    pthread_t th[NB_THREADS];
    struct tranche tranches[NB_THREADS];
    long chunk = N_TRANCHES / NB_THREADS;  // tranches pour chaque thread

    for (int i = 0; i < NB_THREADS; i++) {
        tranches[i].debut = i * chunk;
        tranches[i].fin   = (i == NB_THREADS - 1) ? N_TRANCHES : (i+1) * chunk;
        pthread_create(&th[i], NULL, calculer_tranche, &tranches[i]);
    }

    double total = 0.0;
    for (int i = 0; i < NB_THREADS; i++) {
        pthread_join(th[i], NULL);
        total += tranches[i].resultat;  // accumulation après join (pas de race)
    }

    printf("pi = %.10f\n", 4.0 * total * PAS);
    return 0;
}
ConceptExplication
struct trancheChaque thread a sa propre structure : ses bornes de travail et son résultat local. Pas de partage → pas de race condition.
pthread_createLance un thread qui exécute la fonction donnée avec l'argument donné.
pthread_joinAttendre la fin d'un thread. Analogue à wait() pour les processus.
Accumulation après joinOn additionne les résultats après avoir attendu tous les threads → aucun risque de lire un résultat partiel.

Exercice — Le million incrémenté (race condition)

million.c ⚠️ Race condition — résultat non déterministe

1000 threads incrémentent chacun 1000 fois une variable globale. En théorie : résultat = 1 000 000. En pratique : bien moins, car ++incremente_moi n'est pas atomique — c'est en réalité 3 opérations (LOAD, ADD, STORE) et des threads peuvent se les voler.

⚠️ Race condition : Thread A lit 42, Thread B lit 42, A écrit 43, B écrit 43 → on a perdu 1 incrémentation ! Ce bug est non déterministe et difficile à reproduire.
Correction avec mutex
#include <pthread.h>
#define N 1000

int incremente_moi = 0;
pthread_mutex_t verrou = PTHREAD_MUTEX_INITIALIZER;  // mutex global

void *bosser(void *arg) {
    for (int i = 0; i < N; i++) {
        pthread_mutex_lock(&verrou);    // ← section critique : début
        ++incremente_moi;
        pthread_mutex_unlock(&verrou);  // ← section critique : fin
    }
    return NULL;
}

int main(void) {
    pthread_t th[N];
    for (int i = 0; i < N; i++)
        pthread_create(&th[i], NULL, bosser, NULL);
    for (int i = 0; i < N; i++)
        pthread_join(th[i], NULL);
    printf("incremente_moi = %d\n", incremente_moi);  // maintenant = 1 000 000
    pthread_mutex_destroy(&verrou);
    return 0;
}
FonctionRôle
pthread_mutex_lock(m)Prend le verrou. Si déjà pris par un autre thread → bloque jusqu'à libération.
pthread_mutex_unlock(m)Libère le verrou. Un thread en attente peut alors le prendre.
PTHREAD_MUTEX_INITIALIZERMacro pour initialiser un mutex statique (alternative à pthread_mutex_init).

Exercice — Tâches avec dépendances (parallélisation)

dependance.c ⚠️ Version séquentielle — à paralléliser en respectant les dépendances

5 tâches A, B, C, D, E avec des dépendances. A et B sont indépendantes, C dépend de A et B, D dépend de C, E dépend de D.

Graphe de dépendances : A ──┐ ├──► C ──► D ──► E B ──┘ A et B : lancés simultanément (pthread) C : doit attendre A ET B → sémaphores (valeur initiale 0, post par A et B, wait ×2 par C) D : attend C E : attend D
Correction avec sémaphores
#include <pthread.h>
#include <semaphore.h>

sem_t sem_a_fini, sem_b_fini, sem_c_fini, sem_d_fini;

void *run_a(void *arg) {
    tache_a();
    sem_post(&sem_a_fini);  // signaler la fin de A
    return NULL;
}
void *run_b(void *arg) {
    tache_b();
    sem_post(&sem_b_fini);  // signaler la fin de B
    return NULL;
}
void *run_c(void *arg) {
    sem_wait(&sem_a_fini);  // attendre A
    sem_wait(&sem_b_fini);  // attendre B
    tache_c();
    sem_post(&sem_c_fini);
    return NULL;
}
void *run_d(void *arg) {
    sem_wait(&sem_c_fini);
    tache_d();
    sem_post(&sem_d_fini);
    return NULL;
}
void *run_e(void *arg) {
    sem_wait(&sem_d_fini);
    tache_e();
    return NULL;
}

int main(void) {
    sem_init(&sem_a_fini, 0, 0);  // valeur initiale 0 = personne n'a fini
    sem_init(&sem_b_fini, 0, 0);
    sem_init(&sem_c_fini, 0, 0);
    sem_init(&sem_d_fini, 0, 0);

    pthread_t ta, tb, tc, td, te;
    pthread_create(&ta, NULL, run_a, NULL);
    pthread_create(&tb, NULL, run_b, NULL);
    pthread_create(&tc, NULL, run_c, NULL);  // bloquera sur sem_wait
    pthread_create(&td, NULL, run_d, NULL);
    pthread_create(&te, NULL, run_e, NULL);

    pthread_join(ta, NULL); pthread_join(tb, NULL);
    pthread_join(tc, NULL); pthread_join(td, NULL);
    pthread_join(te, NULL);
    return 0;
}

Exercice — Rendez-vous à 2 (synchronisation)

rendez_vous_a_2_non-sync.c ⚠️ Non synchronisé — les compteurs ne s'attendent pas

Deux threads comptent de 0 à 100. Objectif : à chaque pas, les deux threads doivent se retrouver ensemble — l'un n'avance pas sans l'autre. C'est le problème du rendez-vous à 2.

Correction avec 2 sémaphores (barrière à 2)
#include <semaphore.h>

sem_t sem1, sem2;  // sem1 : th2 attend th1 — sem2 : th1 attend th2

void *compteur1(void *arg) {
    for (int i = 0; i <= 100; i++) {
        prendre_du_temps_pour_compter();
        printf("a : %d\n", i);
        sem_post(&sem1);  // signaler à th2 que th1 est prêt
        sem_wait(&sem2);  // attendre que th2 soit également prêt
    }
    return NULL;
}

void *compteur2(void *arg) {
    for (int i = 0; i <= 100; i++) {
        prendre_du_temps_pour_compter();
        printf("b : %d\n", i);
        sem_post(&sem2);  // signaler à th1 que th2 est prêt
        sem_wait(&sem1);  // attendre th1
    }
    return NULL;
}

int main(void) {
    sem_init(&sem1, 0, 0);  sem_init(&sem2, 0, 0);
    pthread_t th1, th2;
    pthread_create(&th1, NULL, compteur1, NULL);
    pthread_create(&th2, NULL, compteur2, NULL);
    pthread_join(th1, NULL); pthread_join(th2, NULL);
    sem_destroy(&sem1); sem_destroy(&sem2);
    return 0;
}
Fonction sémaphoreComportement
sem_init(&s, 0, val)Initialise s à val. Le 2ème arg 0 = sémaphore local (non partagé entre processus).
sem_wait(&s)Décrémente s. Si s == 0, bloque jusqu'à ce qu'un autre thread fasse sem_post.
sem_post(&s)Incrémente s. Réveille un thread en attente si s était 0.

Exercice — Barrière (chiffres et alphabet)

barriere_non-sync.c ⚠️ Pas de barrière — l'ordre chiffre/lettre n'est pas garanti

N threads, chacun affiche son chiffre puis sa lettre. Objectif : tous les chiffres doivent être affichés avant toute lettre. Solution : une barrière (tous les threads attendent avant de continuer).

Correction avec barrière POSIX
#include <pthread.h>
#define N 5

pthread_barrier_t barriere;

void *chiffres_et_alphabet(void *arg) {
    int i = *(int *)arg;

    printf("%d\n", i);          // phase 1 : afficher le chiffre

    /* BARRIÈRE : attendre que tous les N threads aient affiché leur chiffre */
    pthread_barrier_wait(&barriere);

    printf("%c\n", 'A' + i);  // phase 2 : afficher la lettre
    return NULL;
}

int main(void) {
    pthread_barrier_init(&barriere, NULL, N);  // N threads doivent wait() avant de passer
    pthread_t th[N]; int nums[N];
    for (int i = 0; i < N; i++) {
        nums[i] = i;
        pthread_create(&th[i], NULL, chiffres_et_alphabet, &nums[i]);
    }
    for (int i = 0; i < N; i++) pthread_join(th[i], NULL);
    pthread_barrier_destroy(&barriere);
    return 0;
}
pthread_barrier : Quand N threads ont appelé pthread_barrier_wait, ils sont tous débloqués simultanément. C'est le point de synchronisation collectif.

Exercice — Danseurs (rendez-vous meneur/suiveur)

danseurs_non_sync.c ⚠️ Race condition sur numero_meneur / numero_suiveur

Meneurs et suiveurs doivent se trouver en même temps pour danser ensemble. Sans synchronisation, un meneur peut lire le numéro d'un suiveur qui n'est pas encore prêt (ou l'inverse).

Correction avec 2 sémaphores (rendez-vous classique)
#include <semaphore.h>
#include <pthread.h>

/* Sémaphores de rendez-vous :
 * sem_meneur_pret : le meneur a mis à jour numero_meneur
 * sem_suiveur_pret : le suiveur a mis à jour numero_suiveur */
sem_t sem_meneur_pret, sem_suiveur_pret;
sem_t sem_meneur_peut_repartir, sem_suiveur_peut_repartir;
int numero_suiveur = -1, numero_meneur = -1;
pthread_mutex_t mutex_rv = PTHREAD_MUTEX_INITIALIZER;

void *meneur(void *indice) {
    int mon_indice = *(int *)indice;
    for (;;) {
        se_preparer();
        /* Écrire son numéro et signaler */
        pthread_mutex_lock(&mutex_rv);
        numero_meneur = mon_indice;
        sem_post(&sem_meneur_pret);     // je suis prêt
        sem_wait(&sem_suiveur_pret);    // attendre le suiveur
        int partenaire = numero_suiveur;
        pthread_mutex_unlock(&mutex_rv);
        meneur_danser(mon_indice, partenaire);
        sem_post(&sem_meneur_peut_repartir);
    }
    return NULL;
}
void *suiveur(void *indice) {
    int mon_indice = *(int *)indice;
    for (;;) {
        se_preparer();
        pthread_mutex_lock(&mutex_rv);
        numero_suiveur = mon_indice;
        sem_post(&sem_suiveur_pret);
        sem_wait(&sem_meneur_pret);
        int partenaire = numero_meneur;
        pthread_mutex_unlock(&mutex_rv);
        suiveur_danser(mon_indice, partenaire);
        sem_post(&sem_suiveur_peut_repartir);
    }
    return NULL;
}

Exercice — Space Invaders (calcul + affichage en parallèle)

space_invaders.c ⚠️ Séquentiel — calcul et affichage devraient être en parallèle

Idée : un thread calcule le prochain état de l'écran pendant que l'autre affiche l'état courant (double buffering). Nécessite 2 sémaphores pour la coordination.

Architecture producteur-consommateur
#include <pthread.h>
#include <semaphore.h>

int ecran[2][HAUTEUR][LARGEUR];  // double buffer
int buf_calcul = 0, buf_affich = 1;

sem_t sem_calcul_pret;   // 0 → afficheur attend que le calcul soit fait
sem_t sem_affich_fini;   // 0 → calculateur attend que l'affichage soit fini

void *thread_calcul(void *arg) {
    for (int n = 0; n < NUM_ITER; n++) {
        sem_wait(&sem_affich_fini);   // attendre que l'afficheur ait fini
        /* Calculer le prochain état dans ecran[buf_calcul] */
        microsleep(DELAI);
        /* ─── swap des buffers ─── */
        buf_affich = buf_calcul; buf_calcul = 1 - buf_calcul;
        sem_post(&sem_calcul_pret);   // signaler à l'afficheur
    }
    return NULL;
}

void *thread_affich(void *arg) {
    for (int n = 0; n < NUM_ITER; n++) {
        sem_wait(&sem_calcul_pret);   // attendre le calcul
        /* Afficher ecran[buf_affich] */
        effacer_lignes(HAUTEUR);
        microsleep(DELAI);
        sem_post(&sem_affich_fini);   // signaler la fin de l'affichage
    }
    return NULL;
}

void lire_ecrire_majuscule(int fdin, int fdout) {
    char buffer[BUFSZ];
    ssize_t n;
    while ((n = read(fdin, buffer, BUFSZ)) > 0) {
        for (size_t i = 0; i < n; i++)
            buffer[i] = toupper(buffer[i]);  // conversion char par char
        write(fdout, buffer, n);
    }
    // n == 0 quand le père a fermé tube[1] → EOF → on sort de la boucle
}

int init_sock_ecoute() {
    /* 1. Créer une socket TCP IPv4 */
    int sock = socket(AF_INET, SOCK_STREAM, 0);
    //                          ↑
    //                    TCP (avec connexion)
    if (sock < 0) perror_exit("socket");

    /* Option pour réutiliser l'adresse immédiatement après arrêt du serveur */
    int opt = 1;
    setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    /* 2. Bind sur toutes les interfaces, port 6666 */
    struct sockaddr_in sa = {
        .sin_family      = AF_INET,
        .sin_port        = htons(PORT_QUIZZP),
        .sin_addr.s_addr = htonl(INADDR_ANY)
    };
    if (bind(sock, (struct sockaddr *)&sa, sizeof(sa)) < 0)
        perror_exit("bind");

    /* 3. Écouter (mettre en file d'attente jusqu'à 10 connexions) */
    if (listen(sock, 10) < 0)
        perror_exit("listen");

    printf("Serveur en écoute sur le port %d\n", PORT_QUIZZP);

void echanger_avec_client(int fd, const struct question *q) {

    /* Envoyer le nombre de propositions (uint16_t = 2 octets) */
    uint16_t nb = htons(q->nb_props);

TCP vs UDPDifférence clé
connect() côté clientIndispensable en TCP pour établir la connexion. Pas besoin en UDP.
listen() + accept()Côté serveur TCP : listen() déclare le serveur prêt, accept() bloque jusqu'à une connexion et crée un nouveau fd.
exact_read() vs read()En TCP, read() peut retourner moins d'octets que demandé (fragmentation réseau). exact_read() boucle jusqu'à lire exactement N octets.
sock_écoute vs sock_échangeLe fd retourné par accept() est DISTINCT du fd d'écoute. Le fd d'écoute reste disponible pour de nouveaux clients.
